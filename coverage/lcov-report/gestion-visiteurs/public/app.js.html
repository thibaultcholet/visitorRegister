function startCountdown() {
    const closeButton = document.getElementById('close-popup');
    let timeLeft = 5;
    
    console.log('Dmarrage du countdown avec', timeLeft, 'secondes');
    
    // Fonction pour mettre  jour le bouton
    function updateButton() {
        if (closeButton) {
            closeButton.innerHTML = `
                <i class="fas fa-thumbs-up"></i>
                Parfait ! (${timeLeft})
            `;
        }
    }
    
    // Initialiser le bouton avec le compteur
    updateButton();
    
    // Dmarrer le compte  rebours
    countdownTimer = setInterval(() => {
        timeLeft--;
        console.log('Countdown:', timeLeft);
        
        if (timeLeft > 0) {
            updateButton();
        } else {
            // Temps coul, fermer automatiquement
            console.log('Fin du countdown, fermeture automatique');
            clearInterval(countdownTimer);
            hideSuccessPopup();
        }
    }, 1000);
}
